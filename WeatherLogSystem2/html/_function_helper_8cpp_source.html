192string FunctionHelper::FindsPCCForGivenMonth(map<int, map<int, map<int, Bst<Weather>>>>& windLog, int month)
193{
194 Logic logicHelp;
195 Vector<float> windSpeeds, airTemps, solarRads;
196 string result;
197
198 // Initialize sPCC values for each correlation
199 float spcc_ST_total = 0.0, spcc_SR_total = 0.0, spcc_TR_total = 0.0;
200 int yearCount = 0; // To count the number of years processed
201
202 // Loop through all years in the windLog map
203 for (const auto& yearEntry : windLog)
204 {
205 int year = yearEntry.first;
206 logicHelp.extractMonthData(windLog, month, year, windSpeeds, airTemps, solarRads);
207
208 // Calculate sPCC for each year
209 float spcc_ST = logicHelp.calculateSPCC(windSpeeds, airTemps);
210 float spcc_SR = logicHelp.calculateSPCC(windSpeeds, solarRads);
211 float spcc_TR = logicHelp.calculateSPCC(airTemps, solarRads);
212
213 // Aggregate sPCC values
214 spcc_ST_total += spcc_ST;
215 spcc_SR_total += spcc_SR;
216 spcc_TR_total += spcc_TR;
217 yearCount++;
218 }
219
220 // Calculate average sPCC values
221 float spcc_ST_avg = yearCount > 0 ? spcc_ST_total / yearCount : 0.0;
222 float spcc_SR_avg = yearCount > 0 ? spcc_SR_total / yearCount : 0.0;
223 float spcc_TR_avg = yearCount > 0 ? spcc_TR_total / yearCount : 0.0;
224
225 result += "Average Sample Pearson Correlation Coefficient for month " + logicHelp.IntMonthToString(month) + "\n";
226 result += "S_T: " + to_string(spcc_ST_avg) + "\n";
227 result += "S_R: " + to_string(spcc_SR_avg) + "\n";
228 result += "T_R: " + to_string(spcc_TR_avg) + "\n";
229
230 return result;
231}
